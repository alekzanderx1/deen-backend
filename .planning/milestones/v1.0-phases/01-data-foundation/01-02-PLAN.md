```python
def parse_pdf(pdf_path: str) -> str:
    """Extract full text from PDF, joining all pages with newlines."""
    doc = fitz.open(pdf_path)
    pages = [page.get_text() for page in doc]
    doc.close()
    full_text = "\n".join(pages)
    logger.info("Extracted %d pages from %s", len(pages), pdf_path)
    return full_text
```

**chunk_rulings() function — CRITICAL IMPLEMENTATION:**

The function must:
1. Split full_text on RULING_PATTERN to get alternating [preamble, "Ruling N.", ruling_number_str, body_text, ...] segments
2. Walk the segments linearly, tracking `current_chapter` and `current_section` state that persists across ruling boundaries (Pitfall 5)
3. Before splitting on rulings: scan the full_text top-to-bottom for CHAPTER_PATTERN matches to build a chapter-start-position map
4. For each ruling body: count tokens with `len(ENCODING.encode(text))`
5. If tokens <= MAX_CHUNK_TOKENS: emit one chunk dict
6. If tokens > MAX_CHUNK_TOKENS: apply `TokenTextSplitter(encoding_name="cl100k_base", chunk_size=TARGET_CHUNK_TOKENS, chunk_overlap=0)` to split into sub-chunks; each sub-chunk shares the same ruling_number but gets a different chunk_idx
7. Filter out any chunk where `len(ENCODING.encode(chunk["text"])) < MIN_CHUNK_TOKENS` (Pitfall 4)
8. Vector ID format: `f"fiqh-{ruling_number}-{chunk_idx}"` (per D-13)

Each output dict has exactly these keys:
```python
{
    "id": str,           # e.g. "fiqh-712-0"
    "text": str,         # chunk text, 20-400 tokens
    "source_book": str,  # always "Islamic Laws 4th Edition"
    "chapter": str,      # e.g. "CHAPTER TWO" or "" if before first chapter
    "section": str,      # e.g. "1. Kurr water" or ""
    "ruling_number": int, # e.g. 712
    "topic_tags": list[str],  # e.g. ["tahara"]
}
```

**Chapter/section state tracking approach:**
Build a dict of {character_position: chapter_name} and {character_position: section_name} by scanning the full text once before ruling splitting. When processing each ruling, determine which chapter/section was most recently encountered before that ruling's position.

```python
def chunk_rulings(full_text: str) -> list[dict[str, Any]]:
    """Split text into chunks at ruling boundaries with metadata."""
    # Step 1: Build chapter position map by scanning full text once
    chapter_positions: list[tuple[int, str]] = []
    for m in CHAPTER_PATTERN.finditer(full_text):
        chapter_positions.append((m.start(), m.group(0)))

    section_positions: list[tuple[int, str]] = []
    for m in re.finditer(r'^\d+\.\s+\S[^\n]*', full_text, re.MULTILINE):
        # Only count as section if it's a short line (< 80 chars, not a ruling body)
        line = m.group(0)
        if len(line) < 80:
            section_positions.append((m.start(), line.strip()))

    # Step 2: Split on ruling boundaries
    splitter = TokenTextSplitter(
        encoding_name="cl100k_base",
        chunk_size=TARGET_CHUNK_TOKENS,
        chunk_overlap=0,
    )
    parts = RULING_PATTERN.split(full_text)
    # parts layout: [pre_text, "Ruling N.", "N", body, "Ruling M.", "M", body, ...]

    chunks: list[dict[str, Any]] = []
    position_offset = len(parts[0]) if parts else 0  # skip preamble

    i = 1  # skip preamble at parts[0]
    while i < len(parts) - 2:
        ruling_header = parts[i]       # e.g. "Ruling 712."
        ruling_num_str = parts[i + 1]  # e.g. "712"
        body = parts[i + 2]            # ruling body text
        i += 3

        ruling_number = int(ruling_num_str)
        ruling_text = f"{ruling_header} {body.strip()}"

        # Determine chapter and section from position map
        current_pos = position_offset
        position_offset += len(ruling_header) + len(ruling_num_str) + len(body)

        chapter = ""
        for pos, chap in chapter_positions:
            if pos <= current_pos:
                chapter = chap
        section = ""
        for pos, sec in section_positions:
            if pos <= current_pos:
                section = sec

        topic_tag = assign_topic_tag(chapter)

        # Determine sub-chunks
        token_count = len(ENCODING.encode(ruling_text))
        if token_count <= MAX_CHUNK_TOKENS:
            sub_texts = [ruling_text]
        else:
            sub_texts = splitter.split_text(ruling_text)

        for chunk_idx, sub_text in enumerate(sub_texts):
            if len(ENCODING.encode(sub_text)) < MIN_CHUNK_TOKENS:
                continue  # Filter phantom cross-reference fragments
            chunks.append({
                "id": f"fiqh-{ruling_number}-{chunk_idx}",
                "text": sub_text,
                "source_book": SOURCE_BOOK,
                "chapter": chapter,
                "section": section,
                "ruling_number": ruling_number,
                "topic_tags": [topic_tag],
            })

    logger.info("Produced %d chunks from %d rulings", len(chunks), ruling_number if chunks else 0)
    return chunks
```
